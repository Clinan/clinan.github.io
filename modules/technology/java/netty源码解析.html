# Netty源码解析

# 原生JavaNIO

实现的是单线程Reactor模式的NIO。

/**
 * 最原生的JavaNIO, 了解最原生的JavaNIO是如何工作的。
 */
public class JavaNio {
    public static Selector selector = null;

    public static void main(String[] args) throws IOException {
        ServerSocketChannel ssc = ServerSocketChannel.open();
        SocketAddress socketAddress = new InetSocketAddress(9999);
        selector = Selector.open();
        ssc.configureBlocking(false);
        // ①
        ssc.bind(socketAddress);
        // 注册监听Accept事件
        // ②
        ssc.register(selector, SelectionKey.OP_ACCEPT);

        while (true) {
            /**
             * 说白了，这个就是个阻塞的方法。有数据就会被唤醒
             * 下面的是官方文档翻译
             * <p>
             * 选择一组键，其对应的通道已准备好进行 I/O 操作。
             * 此方法执行阻塞选择操作。 它仅在至少选择一个通道、
             * 调用此选择器的唤醒方法或当前线程被中断（以先到者为准）后返回。
             */
            int op = selector.select();
            System.out.println("OPS:" + op);
            if (op > 0) {
                Iterator<SelectionKey> iterator = selector.selectedKeys().iterator();
                // 迭代有数据的数组，然后交由给线程处理
                while (iterator.hasNext()) {
                    SelectionKey event = iterator.next();
                    iterator.remove();
                    if (event.isAcceptable()) {
                        // 这是accept事件
                        ServerSocketChannel ssc2 = (ServerSocketChannel) event.channel();
                        // ③
                        SocketChannel sc = ssc2.accept();
                        // 这个很重要，不然会阻塞当前线程
                        sc.configureBlocking(false);
                        // 注册 read事件
                        //  ④
                        sc.register(selector, SelectionKey.OP_READ);
                    } else if (event.isReadable()) {
                        // 这是已建立的socket连接，发送数据的事件。即有数据可读==菜做好了，可以去端了。
                        SocketChannel channel = (SocketChannel) event.channel();
                        ByteBuffer buffer = ByteBuffer.allocate(1024);
                        // ⑤
                        channel.read(buffer);
                        //调整指针位置
                        buffer.flip();
                        System.out.println(Arrays.toString(buffer.array()));
                    }
                }
            }
        }
    }
}

在接下来，我们就根据上面的关键节点①②③④⑤进行查看源代码

  1. 如何绑定端口，bind
  2. 监听端口, register
  3. 如何创建socket连接，和客户端连接上，accept
  4. SocketChannel和selector绑定，register
  5. 如何读取数据。read
  6. 其实还有如何返回数据给客户端。

# 替换SelectorImpl中的属性

private SelectorTuple openSelector() {
    final Selector unwrappedSelector;
    try {
        // 调用JDK方法，创建一个Selector的实例SelectorImpl对象
        unwrappedSelector = provider.openSelector();
    } catch (IOException e) {
        throw new ChannelException("failed to open a new selector", e);
    }
	// 这是一个选项
    if (DISABLE_KEY_SET_OPTIMIZATION) {
        return new SelectorTuple(unwrappedSelector);
    }

    Object maybeSelectorImplClass = AccessController.doPrivileged(new PrivilegedAction<Object>() {
        @Override
        public Object run() {
            try {
                return Class.forName(
                        "sun.nio.ch.SelectorImpl",
                        false,
                        PlatformDependent.getSystemClassLoader());
            } catch (Throwable cause) {
                return cause;
            }
        }
    });

    if (!(maybeSelectorImplClass instanceof Class) ||
            // ensure the current selector implementation is what we can instrument.
            !((Class<?>) maybeSelectorImplClass).isAssignableFrom(unwrappedSelector.getClass())) {
        if (maybeSelectorImplClass instanceof Throwable) {
            Throwable t = (Throwable) maybeSelectorImplClass;
            logger.trace("failed to instrument a special java.util.Set into: {}", unwrappedSelector, t);
        }
        return new SelectorTuple(unwrappedSelector);
    }
    final Class<?> selectorImplClass = (Class<?>) maybeSelectorImplClass;
    // 创建自定义的selectorKeySet
    final SelectedSelectionKeySet selectedKeySet = new SelectedSelectionKeySet();

    Object maybeException = AccessController.doPrivileged(new PrivilegedAction<Object>() {
        @Override
        public Object run() {
            try {
                Field selectedKeysField = selectorImplClass.getDeclaredField("selectedKeys");
                Field publicSelectedKeysField = selectorImplClass.getDeclaredField("publicSelectedKeys");
                //... 省略部分代码
				// 通过反射赋值给selector对象
                selectedKeysField.set(unwrappedSelector, selectedKeySet);
                publicSelectedKeysField.set(unwrappedSelector, selectedKeySet);
                return null;
            } catch (NoSuchFieldException e) {
                return e;
            } catch (IllegalAccessException e) {
                return e;
            }
        }
    });

    if (maybeException instanceof Exception) {
        selectedKeys = null;
        Exception e = (Exception) maybeException;
        logger.trace("failed to instrument a special java.util.Set into: {}", unwrappedSelector, e);
        return new SelectorTuple(unwrappedSelector);
    }
    // 将上面new出来的selectedKeySet赋值
    selectedKeys = selectedKeySet;
    logger.trace("instrumented a special java.util.Set into: {}", unwrappedSelector);
    return new SelectorTuple(unwrappedSelector,
            new SelectedSelectionKeySetSelector(unwrappedSelector, selectedKeySet));
}

# 自定义的SelectionKeySet

主要是用数组实现了一个Set接口。因为HashSet的时间复杂度在hash冲突的时候是O(n)，而数组永远是O(1)。常数项上的操作也会比HashSet少，数组扩容也更容易。

final class SelectedSelectionKeySet extends AbstractSet<SelectionKey> {
    SelectionKey[] keys;
    int size;
    SelectedSelectionKeySet() {
        keys = new SelectionKey[1024];
    }
    @Override
    public boolean add(SelectionKey o) {
        if (o == null) {
            return false;
        }
        keys[size++] = o;
        if (size == keys.length) {
            increaseCapacity();
        }
        return true;
    }
    
    @Override
    public int size() {
        return size;
    }
    void reset() {
        reset(0);
    }

    void reset(int start) {
        Arrays.fill(keys, start, size, null);
        size = 0;
    }

    private void increaseCapacity() {
        SelectionKey[] newKeys = new SelectionKey[keys.length << 1];
        System.arraycopy(keys, 0, newKeys, 0, size);
        keys = newKeys;
    }
}
